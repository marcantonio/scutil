def test_alter_options
  divert_stdout
  @exec.exec_command(TRUE_COMMAND)
  revert_stdout
  assert_not_match(/\[#{TestScutil.hostname}\]/, @output.string)
  
  @exec.set_options({ :scutil_verbose => true })
  
  divert_stdout
  @exec.exec_command(TRUE_COMMAND)
  revert_stdout
  assert_match(/\[#{TestScutil.hostname}\]/, @output.string)
end
