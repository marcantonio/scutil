def test_exec_command_output
  divert_stdout
  Scutil.exec_command(TestScutil.hostname, TestScutil.user, 'echo "alpha"', nil, { :port => TestScutil.port })
  revert_stdout
  assert_equal "alpha", @output.string.chomp
end
